  const total=Math.max(1,imgs.length);
  function bump(){ done++; const p=Math.round(done/total*100); if(lbar)lbar.style.width=p+'%'; if(lsub)lsub.textContent='steeping · '+p+'%'; if(done>=total) finish(); }
  let finished=false;
  function finish(){ if(finished)return; finished=true; setTimeout(()=>{ loader.classList.add('done'); start(); }, 350); }
  imgs.forEach(im=>{ if(im.complete) bump(); else { im.addEventListener('load',bump); im.addEventListener('error',bump); } });
  setTimeout(finish, 6000); // safety

  /* ---- nav dock ---- */
  const nav=document.getElementById('nav');
  addEventListener('scroll',()=>{ nav.classList.toggle('docked', (window.scrollY||0) > 60); }, {passive:true});

  /* ---- hero steam canvas (light, pre-rendered sprites) ---- */
  (function(){
    const cv=document.getElementById('steam'); if(!cv) return; const x=cv.getContext('2d');
    let W=0,H=0, DPR=1, parts=[], spr, running=false, raf=0, lastT=0;
    function sprite(){ const s=128,c=document.createElement('canvas'); c.width=c.height=s; const g=c.getContext('2d');
      const gr=g.createRadialGradient(s/2,s/2,0,s/2,s/2,s/2); gr.addColorStop(0,'rgba(244,240,230,.5)'); gr.addColorStop(1,'rgba(244,240,230,0)');
      g.fillStyle=gr; g.fillRect(0,0,s,s); return c; }
    function size(){ const r=cv.getBoundingClientRect(); W=cv.width=Math.max(2,r.width*DPR); H=cv.height=Math.max(2,r.height*DPR); }
    function seed(){ parts=[]; const n=Math.min(14,Math.round(W/130)); for(let i=0;i<n;i++) parts.push(mk()); }
    function mk(){ return {x:Math.random()*W, y:H*(.5+Math.random()*.6), r:(30+Math.random()*60)*DPR, v:(.15+Math.random()*.35)*DPR, a:0, av:.002+Math.random()*.004, drift:(Math.random()-.5)*.3*DPR, ph:Math.random()*6.28}; }
    function frame(t){ raf=requestAnimationFrame(frame); if(!running) return; if(t-lastT<33) return; lastT=t; x.clearRect(0,0,W,H); x.globalCompositeOperation='screen';
      for(const p of parts){ p.y-=p.v; p.x+=p.drift+Math.sin(t*0.0004+p.ph)*0.25*DPR; p.a+=p.av; const life=Math.max(0,Math.min(1,Math.sin(p.a)));
        if(p.y< H*0.18 || p.a>3.14){ Object.assign(p, mk()); continue; }
        const s=p.r*(1+p.a*0.35); x.globalAlpha=life*0.5; x.drawImage(spr, p.x-s/2, p.y-s/2, s, s); }
      x.globalAlpha=1; }
    spr=sprite(); size(); seed();
    addEventListener('resize',()=>{ size(); seed(); });
    try{ new IntersectionObserver(es=>{ running=es[0].isIntersecting && !REDUCED; },{threshold:.02}).observe(cv); }catch(e){ running=!REDUCED; }
    if(!REDUCED){ running=true; raf=requestAnimationFrame(frame); }
  })();

  /* ---- everything below needs GSAP ---- */
  function start(){

    // Lenis ↔ GSAP share one clock so pinned scrubs stay in sync
    const lenis = (!REDUCED && window.Lenis) ? new Lenis({lerp:0.07, smoothWheel:true, wheelMultiplier:1}) : null;
    window.lenis = lenis;
    if(lenis){ lenis.on('scroll', ScrollTrigger.update); gsap.ticker.add(t=>lenis.raf(t*1000)); gsap.ticker.lagSmoothing(0); }

    // anchor links via lenis
    document.querySelectorAll('a[href^="#"]').forEach(a=>a.addEventListener('click',e=>{

    }));

    if(REDUCED){

      return;
    }

    // hero parallax

      gsap.fromTo(el,{opacity:0,y:52},{opacity:1,y:0,duration:1.25,ease:'power3.out',
